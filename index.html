// ===================================================== // COLE TODO ESTE CÓDIGO NO APPS SCRIPT E REIMPLANTE // ===================================================== const SHEET_NAME = "BENFEITORIAS MCM"; function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.getSheets()[0]; const data = sheet.getDataRange().getValues(); // Encontra a linha de cabeçalhos (procura "CASA") let headerIdx = 2; for (let i = 0; i < Math.min(data.length, 6); i++) { if (data[i].some(c => String(c).toUpperCase().includes('CASA'))) { headerIdx = i; break; } } const headers = data[headerIdx]; const rows = []; for (let i = headerIdx + 1; i < data.length; i++) { if (!data[i][0] && !data[i][1]) continue; const obj = {}; headers.forEach((h, j) => { const v = data[i][j]; // Formata datas e números como string legível if (v instanceof Date) { const mm = String(v.getMonth()+1).padStart(2,'0'); const yyyy = v.getFullYear(); obj[h] = mm + '/' + yyyy; } else if (typeof v === 'number' && String(h).toUpperCase().includes('GASTO')) { obj[h] = 'R$ ' + v.toLocaleString('pt-BR', {minimumFractionDigits:2}); } else { obj[h] = v; } }); obj._row = i + 1; rows.push(obj); } const json = JSON.stringify({status:"ok", data:rows}); const callback = (e.parameter || {}).callback; if (callback) { return ContentService.createTextOutput(callback + "(" + json + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.getSheets()[0]; // Aceita tanto application/json quanto text/plain const raw = e.postData ? e.postData.contents : '{}'; const payload = JSON.parse(raw); if (payload.action === "update") { const row = payload._row; sheet.getRange(row, 1).setValue(payload.CASA || ""); sheet.getRange(row, 2).setValue(payload["OBSERVAÇÃO"] || ""); sheet.getRange(row, 3).setValue(payload.CATEGORIA || ""); sheet.getRange(row, 4).setValue(payload.DEPARTAMENTO || ""); sheet.getRange(row, 5).setValue(payload["PREV. DE INICIO"] || ""); sheet.getRange(row, 6).setValue(payload.STATUS || ""); sheet.getRange(row, 7).setValue(payload["PREVISÃO DE GASTO"] || ""); sheet.getRange(row, 8).setValue(payload.Prioridade || ""); return ContentService.createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } if (payload.action === "insert") { sheet.appendRow([payload.CASA, payload["OBSERVAÇÃO"], payload.CATEGORIA, payload.DEPARTAMENTO, payload["PREV. DE INICIO"], payload.STATUS, payload["PREVISÃO DE GASTO"], payload.Prioridade]); return ContentService.createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } if (payload.action === "delete" && payload._row) { sheet.deleteRow(payload._row); return ContentService.createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({status:"error"})) .setMimeType(ContentService.MimeType.JSON); }
